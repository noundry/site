Noundry.Tuxedo

Modern ORM for .NET

Built on Dapper's proven performance with CRUD operations for SQL Server, PostgreSQL, MySQL, and SQLite. High-performance, lightweight, and developer-friendly.

  • Built-in async CRUD operations
  • Convention-based mapping
  • Resiliency & Transaction support
  • Database-Specific Custom Exceptions
  • Entity Change Auditing
  • Clean POCOs with attributes
ProductService.cs
// Configure in Program.cs
builder.Services.AddTuxedoSqlServer(connectionString);

// Use in your services
public class ProductService
{
    private readonly IDbConnection _connection;

    public async Task<Product> GetProductAsync(int id)
    {
        return await _connection.GetAsync<Product>(id);
    }

    public async Task<int> CreateProductAsync(Product product)
    {
        return await _connection.InsertAsync(product);
    }

    public async Task<IEnumerable<Product>> GetAllAsync()
    {
        return await _connection.GetAllAsync<Product>();
    }
}

Noundry.Tuxedo.Bowtie

Schema Synchronization

Database schema synchronization for Tuxedo. Define schema directly in POCO model classes with attributes, and Bowtie keeps your database in perfect sync.

  • Model-first schema sync
  • Attribute-driven configuration
  • Indexes, constraints, relationships
Product.cs
// Configure services
builder.Services.AddTuxedoSqlServer(connectionString);
builder.Services.AddNoundryTuxedoBowtie();

// Sync database on startup
await app.Services.SynchronizeDatabaseAsync();

// Define schema with attributes
[Table("Products")]
public class Product
{
    [Key]
    public int Id { get; set; }

    [Column(MaxLength = 100)]
    [Index("IX_Products_Name")]
    public string Name { get; set; }

    [CheckConstraint("Price > 0")]
    public decimal Price { get; set; }

    [ForeignKey("Categories", "Id")]
    public int CategoryId { get; set; }
}

// Configure OAuth providers
builder.Services.AddAuthnz(options =>
{
    options.AddGoogle(clientId, clientSecret);
    options.AddMicrosoft(clientId, clientSecret);
    options.AddGitHub(clientId, clientSecret);
    options.JwtSecret = jwtSecret;
});

// Use in controllers
[HttpGet("/auth/google")]
public IActionResult GoogleLogin()
{
    var url = _authnz.GetAuthorizationUrl("google");
    return Redirect(url);
}

[HttpGet("/auth/callback")]
public async Task<IActionResult> Callback(string code)
{
    var result = await _authnz.ExchangeCodeAsync(code);
    var jwt = _authnz.GenerateJwtToken(result.User);
    return Ok(new { Token = jwt });
}

UserService.cs
public class UserService
{
    public User CreateUser(string email, int age)
    {
        // Guard against null or empty
        email.Guard().NotNullOrEmpty(nameof(email));

        // Guard against invalid email
        email.Guard().Email(nameof(email));

        // Guard against invalid range
        age.Guard().InRange(18, 120, nameof(age));

        return new User { Email = email, Age = age };
    }

    public Product GetProduct(int id, Product product)
    {
        // Guard against negative values
        id.Guard().NotNegative(nameof(id));

        // Guard against null objects
        product.Guard().NotNull(nameof(product));

        return product;
    }
}

HTTP client wrapper with retry policies, circuit breakers, and automatic logging. Build resilient microservices with ease.

  • Automatic retry with exponential backoff
  • Circuit breaker pattern
  • Request/response logging
ExternalApiService.cs
// Configure in Program.cs
builder.Services.AddConnector("ApiClient", options =>
{
    options.BaseUrl = "https://api.example.com";
    options.RetryCount = 3;
    options.TimeoutSeconds = 30;
    options.CircuitBreakerThreshold = 5;
});

// Use in services
public class ExternalApiService
{
    private readonly IConnector _connector;

    public async Task<Product> GetProductAsync(int id)
    {
        return await _connector
            .GetAsync<Product>($"/products/{id}");
    }

    public async Task CreateOrderAsync(Order order)
    {
        await _connector.PostAsync("/orders", order);
    }
}
