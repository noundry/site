app/routes/index.tsx
// app/routes/index.tsx
import { Hono } from 'hono'
import { z } from 'zod'

const app = new Hono()

// Define schema
const userSchema = z.object({
  email: z.string().email(),
  name: z.string().min(2)
})

// Create API route
app.post('/api/users', async (c) => {
  const data = await c.req.json()
  const user = userSchema.parse(data)
  return c.json(user)
})

export default app
