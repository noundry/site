REST API Template

C#

Complete REST API with CRUD operations, validation, error handling, and Swagger documentation. 

[ApiController]
[Route("api/[controller]")]
public class ProductsController : ControllerBase
{
    private readonly IRepository<Product> _repository;

    [HttpGet]
    public async Task<ActionResult<IEnumerable<Product>>> GetAll()
    {
        var products = await _repository.GetAllAsync();
        return Ok(products);
    }

    [HttpPost]
    public async Task<ActionResult<Product>> Create(Product product)
    {
        await _repository.AddAsync(product);
        return CreatedAtAction(nameof(GetById), new { id = product.Id }, product);
    }
}

Repository Pattern

C#

Generic repository with specifications, unit of work, and Tuxedo ORM integration. 

public class Repository<T> : IRepository<T> where T : class
{
    private readonly IDbConnection _connection;

    public async Task<T> GetByIdAsync(int id)
    {
        return await _connection.GetAsync<T>(id);
    }

    public async Task<IEnumerable<T>> FindAsync(ISpecification<T> spec)
    {
        return await _connection.QueryAsync<T>(spec.ToSql(), spec.Parameters);
    }
}

Background Jobs

C#

Hosted services and background workers with scoped dependencies and error handling. 

public class EmailWorker : BackgroundService
{
    protected override async Task ExecuteAsync(CancellationToken stoppingToken)
    {
        while (!stoppingToken.IsCancellationRequested)
        {
            using var scope = _services.CreateScope();
            var queue = scope.ServiceProvider.GetRequiredService<IEmailQueue>();

            await queue.ProcessPendingEmailsAsync();
            await Task.Delay(TimeSpan.FromMinutes(1), stoppingToken);
        }
    }
}
