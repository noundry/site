Noundry.AIG - AI Gateway

Unified AI Gateway for .NET developers - Access OpenAI, Anthropic, Google Gemini, and more through a single, consistent API with automatic failover, streaming support, and powerful prompt building capabilities.

Installation

$ dotnet add package Noundry.AIG.Client

Quick Start

using Noundry.AIG.Client;
using Noundry.AIG.Client.Builders;
using Noundry.AIG.Client.Configuration;
using Noundry.AIG.Core.Extensions;
using Noundry.AIG.Core.Models;
using Noundry.AIG.Providers;

// Configure the client
var options = new AigClientOptions
{
    UseLocalProviders = true,
    ProviderConfigs = new Dictionary<string, ProviderConfig>
    {
        ["openai"] = new ProviderConfig
        {
            ApiKey = Environment.GetEnvironmentVariable("OPENAI_API_KEY")
        }
    }
};

var aigClient = new AigClient(new ProviderFactory(httpClientFactory), options);

// Send a request
var prompt = new PromptBuilder()
    .WithModel("openai/gpt-4")
    .WithTemperature(0.7f)
    .AddSystemMessage("You are a helpful assistant.")
    .AddUserMessage("Explain quantum computing.")
    .Build();

var response = await aigClient.SendAsync(prompt);
Console.WriteLine(response.GetTextContent());

PromptBuilder Example

var prompt = new PromptBuilder()
    .WithModel("anthropic/claude-sonnet-4")
    .WithTemperature(0.8f)
    .WithMaxTokens(2000)
    .AddSystemMessage("You are a creative writer.")
    .AddUserMessage("Write a story opening.")
    .Build();

var response = await aigClient.SendAsync(prompt);

ChainPromptBuilder Example

var chain = new ChainPromptBuilder()
    .WithDefaultModel("anthropic/claude-sonnet-4")
    .AddStep("Generate", _ =>
        new PromptBuilder()
            .AddUserMessage("Give me a topic."))
    .AddStep("Write", prev =>
        new PromptBuilder()
            .AddUserMessage($"Write about: {prev}"))
    .AddStep("Translate", prev =>
        new PromptBuilder()
            .AddUserMessage($"Translate: {prev}"));

var result = await chain.ExecuteAsync(aigClient);
Console.WriteLine(result.FinalOutput);

Multi-Provider Fallback

// Try multiple providers in order
var prompt = new PromptBuilder()
    .WithModels(
        "openai/gpt-4",
        "anthropic/claude-sonnet-4",
        "google/gemini-pro"
    )
    .AddUserMessage("Generate a story idea.")
    .Build();

var response = await aigClient.SendMultiAsync(prompt);

if (response.HasSuccess)
{
    Console.WriteLine($"Response from: {response.FirstSuccess.Provider}");
    Console.WriteLine(response.FirstSuccess.GetTextContent());
}

Streaming Example

var prompt = new PromptBuilder()
    .WithModel("openai/gpt-4")
    .WithStreaming(true)
    .AddUserMessage("Write a story.")
    .Build();

await foreach (var chunk in aigClient.SendStreamAsync(prompt))
{
    Console.Write(chunk.GetTextContent());
}
