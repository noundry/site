        <noundry-text-input
            asp-for="Email"
            label="Email"
            type="email"
            placeholder="you@example.com"
            required="true" />

        <noundry-text-input
            asp-for="Password"
            label="Password"
            type="password"
            required="true" />

            label="Remember me" />

            type="submit"
            variant="primary"
